create database News_Summary; -- 데이터베이스 생성

use News_Summary; -- 데이터베이스 실행

create table Press( -- 언론사 테이블
   p_id int AUTO_INCREMENT PRIMARY KEY, -- 언론사 아이디
    p_name varchar(10) UNIQUE not null -- 언론사 이름
);

create table N_category( -- 카테고리 대분류
   c_id int AUTO_INCREMENT PRIMARY KEY, -- 대분류 아이디 ex 정치, 경제, 사회, 생활/문화, it/과학, 세계
    c_name varchar(5) UNIQUE not null
);

create table N_category_detail( -- 카테고리 중분류 ex 정치 안에 청화대 국회/정당 북한 행정 국방/외교, 정치일반
   cd_id int AUTO_INCREMENT PRIMARY KEY,
    c_id int not null,
    cd_name varchar(10) unique not null,
   constraint FK_CATEGORY_ID foreign key (c_id) REFERENCES N_category(c_id)
);

create table news( -- 뉴스 핵심 내용들
   n_id int AUTO_INCREMENT PRIMARY KEY, -- news key
    p_id int not null, -- 언론사 번호
    cd_id int not null, -- 중분류 아이디
    n_title varchar(1024) not null, -- news 제목
    n_summary text, -- 요약 내용
    n_input datetime not null, -- news 입력 날짜 및 시간
    n_update datetime, -- news 수정 날짜 및 시간 쓰던가 말던가
    constraint FK_CATEGORY_DT_ID foreign key (cd_id) REFERENCES N_category_detail(cd_id),
    constraint FK_PRESS_ID foreign key (p_id) REFERENCES Press(p_id)
);

create table news_detail(
   nd_id  int AUTO_INCREMENT PRIMARY KEY,
    n_id int,
    nd_description MEDIUMTEXT,
    nd_img varchar(1024)
);
